Table Customers {
  cpf varchar [PK]
  name varchar
}

Table Products {
  id int [PK]
  description varchar
  price float
  created_at datetime
}

Table Products_Prices{
  id int [pk]
  produt_id int [ref: > Products.id]
  value float
  started_at datetime
  finished_at datetime
}

Table Carts_Products{
  cart_id int [pk, ref: > Carts.id]
  product_fk int [pk,ref: > Products.id]
  quantity int
  discount float
  price float
}

Table Carts {
  id int [pk]
  customer_fk varchar [pk, ref: > Customers.cpf]
  created_at datetime
}

Table Orders {
  id int [pk]
  cart_id int [pk, ref: > Carts.id]
  discount int
  created_at datetime
}
